		int [] a = {5,30,51,66,96,77,42,38,36,35,95,42,35,68,98};
		List[] hashTable = new List[31];

		for(int i=0;i<hashTable.length;i++){
			hashTable[i] = new List("index "+i);
		}

		hashTable[a[0]%31].insertAtFront(a[0]);
