Week 9 Ex. 5: Write a filter that selects the words starting with 'a' or 'b'. How big is it?.

Output:



There are 1318 words in this file

Source Code:

package CSIS2420unit2;

import java.io.*;
import java.util.*;

public class Filter_AB {
	public static void main (String[] args) throws IOException{
		String [] wordArray= utilities.fileToStringArray("awmt.txt");
		List wordList = Arrays.asList(wordArray);
		ArrayList al = new ArrayList(wordList);
		
		Iterator it = al.iterator();
		
		while(it.hasNext()){
			if(!cond(it.next())){
				it.remove();
			}
		}
		
		for(String s:al){
			System.out.println(s);
		}
		System.out.println("al now has this many elements: "+al.size());
	}
	
	public static boolean cond (String str){
		str.toLowerCase();
		return (str.charAt(0)=='a')||(str.charAt(0)=='b');
	}
}
