public class Stack2{
	private List s;

	public Stack2() {
		s = new List("The Stack");
	}
	
	public void push (Object obj) {
		s.insertAtFront(obj);
	}
	
	public Object pop () {
		return s.removeFromFront();
	}
	
	public boolean isEmpty() {
		return s.isEmpty();
	}
	
	public String print () {
		return s.print();
	}
}
